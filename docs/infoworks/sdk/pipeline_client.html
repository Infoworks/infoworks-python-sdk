 14    def poll_job(self, pipeline_id=None, job_id=None, poll_timeout=local_configurations.POLLING_TIMEOUT,
 15                 polling_frequency=local_configurations.POLLING_FREQUENCY_IN_SEC,
 16                 retries=local_configurations.NUM_POLLING_RETRIES):
 17        failed_count = 0
 18        response = {}
 19        timeout = time.time() + poll_timeout
 20        while True:
 21            if time.time() > timeout:
 22                break
 23            try:
 24                self.logger.info(f"Failed poll job status count: {failed_count}")
 25                job_monitor_url = url_builder.get_job_status_url(self.client_config, job_id)
 26                response = IWUtils.ejson_deserialize(self.call_api("GET", job_monitor_url,
 27                                                                   IWUtils.get_default_header_for_v3(
 28                                                                       self.client_config['bearer_token'])).content)
 29                result = response.get('result', {})
 30                if len(result) != 0:
 31                    job_status = result["status"]
 32                    self.logger.info(
 33                        "Job poll status : " + result["status"] + "Job completion percentage: " + str(result.get(
 34                            "percentage", 0)))
 35                    if job_status in ["completed", "failed", "aborted"]:
 36                        if job_status == "completed":
 37                            return PipelineResponse.parse_result(status=Response.Status.SUCCESS,
 38                                                                 pipeline_id=pipeline_id,
 39                                                                 job_id=job_id)
 40                        else:
 41                            return PipelineResponse.parse_result(status=job_status, pipeline_id=pipeline_id,
 42                                                                 job_id=job_id)
 43                else:
 44                    self.logger.error(f"Error occurred during job {job_id} status poll")
 45                    if failed_count >= retries - 1:
 46                        return PipelineResponse.parse_result(status=Response.Status.FAILED,
 47                                                             error_code=ErrorCode.GENERIC_ERROR,
 48                                                             error_desc=response, job_id=job_id,
 49                                                             pipeline_id=pipeline_id)
 50                    failed_count = failed_count + 1
 51            except Exception as e:
 52                self.logger.exception("Error occurred during job status poll")
 53                self.logger.info(str(e))
 54                if failed_count >= retries - 1:
 55                    # traceback.print_stack()
 56                    print(response)
 57                    raise PipelineError(response.get("message", "Error occurred during job status poll"))
 58                failed_count = failed_count + 1
 59            time.sleep(polling_frequency)
