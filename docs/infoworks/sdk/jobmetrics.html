326    def get_source_jobs_metrics_results(self, date_string, source, workflow_id=None, workflow_run_id=None):
327        src_name = source["name"]
328        list_of_jobs_obj = self.get_jobs_of_single_source(source["id"], date_string)
329        try:
330            for job in list_of_jobs_obj:
331                job_id = job["id"]
332                job_type = job["type"]
333                job_status = job["status"]
334                job_createdAt = job["created_at"]
335                running_job_template = {
336                    'workflow_id': job.get('triggered_by', {}).get('parent_id', ''),
337                    'workflow_run_id': job.get('triggered_by', {}).get('run_id', ''),
338                    'job_id': job_id,
339                    'job_type': self.job_type_mappings[job_type.upper()],
340                    'job_start_time': job_createdAt.split('.')[0].replace('T', ' '),
341                    'job_end_time': "",
342                    "source_file_names": [],
343                    'job_status': job_status.upper(),
344                    'entity_type': "source",
345                    "source_name": src_name, "source_schema_name": job.get("source_schema_name", ""),
346                    "source_database_name": job.get("source_database_name", ""), "table_group_name": "",
347                    "iwx_table_name": "", 'starting_watermark_value': '', 'ending_watermark_value': '',
348                    "pre_target_count": "", "fetch_records_count": 0,
349                    "target_records_count": ""}
350                running_od = OrderedDict()
351                if job_status.upper() in ["RUNNING", "PENDING", "CANCELED", "ABORTED", "FAILED"]:
352                    for key in ['workflow_id', 'workflow_run_id', 'job_id', 'entity_type', 'job_type', 'job_start_time',
353                                'job_end_time', 'job_status',
354                                'source_name', 'source_file_names', 'source_schema_name', 'source_database_name',
355                                'table_group_name',
356                                'iwx_table_name', 'starting_watermark_value', 'ending_watermark_value',
357                                'target_schema_name', 'target_table_name', 'pre_target_count', 'fetch_records_count',
358                                'target_records_count']:
359                        running_od[key] = running_job_template.get(key, "")
360                    self.job_metrics_final.append(running_od)
361                    continue
362                try:
363                    tg_id, processedAt, job_end_time, job_status, entity_type = self.get_tablegroup_id_from_job(
364                        job_id,
365                        source["id"])
366                    table_group_name, all_tables_list = self.get_table_group_name(tg_id, source["id"])
367                except:
368                    # This means the job is non-tablegroup job
369                    table_group_name = ""
370                    entity_type = "TABLE"
371                    all_tables_list = []
372                ing_metrics = self.get_ingestion_metrics(str(job_id), source["id"])
373                successful_tables = []
374                if ing_metrics != [] and job_type != "export_data":
375                    tables_list = list(set([i['table_id'] for i in ing_metrics]))
376                    df = pd.DataFrame(ing_metrics)
377                    for table_id in tables_list:
378                        successful_tables.append(table_id)
379                        filter1 = df["table_id"] == table_id
380                        table = {}
381                        if len(df.loc[filter1]) > 1:
382                            # Incremental Job
383                            filter2 = df["job_type"] == "CDC"
384                            filter3 = df["job_type"] == "MERGE"
385                            cdc_output = df.loc[filter1 & filter2].to_dict('records')[0]
386                            merge_output = df.loc[filter1 & filter3].to_dict('records')[0]
387                            for item in ['source_id', 'fetch_records_count', 'job_id', 'job_start_time',
388                                         'source_schema_name', 'source_database_name']:
389                                table[item] = cdc_output.get(item, "")
390                                table['job_type'] = "INCREMENTAL"
391                            for item in ['workflow_id', 'workflow_run_id', 'job_end_time', 'job_status',
392                                         'target_records_count', 'first_merged_watermark', 'last_merged_watermark']:
393                                table[item] = merge_output.get(item, "")
394                                table['job_type'] = "INCREMENTAL"
395                        else:
396                            table = df.loc[filter1].to_dict('records')[0]
397                        table_name_url = self.get_table_info(table.get('source_id'), table_id)
398                        table_name = table_name_url.get('name')
399                        target_table_name = table_name_url.get("configuration", {}).get('target_table_name', '')
400                        target_schema_name = table_name_url.get("configuration", {}).get('target_schema_name', '')
401                        table["source_name"] = src_name
402                        table["source_schema_name"] = table.get("source_schema_name", "")
403                        table["source_database_name"] = table.get("source_database_name", "")
404                        table["source_file_names"] = self.get_source_file_paths(source['id'], table_id,
405                                                                                job_id) if table[
406                                                                                               "source_schema_name"] == "" else []
407                        table["table_group_name"] = table_group_name
408                        table["iwx_table_name"] = table_name
409                        table["target_schema_name"] = target_schema_name
410                        table["target_table_name"] = target_table_name
411                        table["workflow_id"] = table.get("workflow_id", "")
412                        table["workflow_run_id"] = table.get('workflow_run_id', "")
413                        if workflow_id is not None and workflow_run_id is not None:
414                            if not (table["workflow_id"] == workflow_id and table[
415                                "workflow_run_id"] == workflow_run_id):
416                                continue
417                        table["entity_type"] = entity_type
418                        table["starting_watermark_value"] = table.pop('first_merged_watermark', '')
419                        table["ending_watermark_value"] = table.pop('last_merged_watermark', '')
420                        if math.isnan(table.get('target_records_count')):
421                            table['pre_target_count'] = None
422                            table['target_records_count'] = None
423                        else:
424                            if table["job_status"] == "FAILED":
425                                table['pre_target_count'] = table.get('target_records_count')
426                                table['target_records_count'] = int(table.get('target_records_count'))
427                            else:
428                                table['pre_target_count'] = int(
429                                    table.get('target_records_count') - int(table.get('fetch_records_count')))
430                                table['target_records_count'] = int(table.get('target_records_count'))
431                        if table.get('job_type') == "CDC":
432                            table['job_type'] = "INCREMENTAL"
433                        od = OrderedDict()
434                        table['job_start_time'] = table['job_start_time'].split('.')[0].replace('T', ' ')
435                        table['job_end_time'] = table['job_end_time'].split('.')[0].replace('T', ' ')
436                        table['fetch_records_count'] = int(table['fetch_records_count'])
437                        for key in ['workflow_id', 'workflow_run_id', 'job_id', 'entity_type', 'job_type',
438                                    'job_start_time',
439                                    'job_end_time', 'job_status',
440                                    'source_name', 'source_file_names', 'source_schema_name', 'source_database_name',
441                                    'table_group_name',
442                                    'iwx_table_name', 'starting_watermark_value', 'ending_watermark_value',
443                                    'target_schema_name', 'target_table_name', 'pre_target_count',
444                                    'fetch_records_count',
