368        """
369        Polls Infoworks Data workflow for given workflow id and run id
370        :param workflow_run_id: run id of the workflow running
371        :type workflow_run_id: String
372        :param workflow_id: entity id of the workflow running
373        :type workflow_id: String
374        :param poll_interval:interval in seconds between poll(default 30)
375        :type poll_interval : Int
376        :return: response dict
377        """
378        response = None
379        try:
380            if workflow_run_id is None:
381                self.logger.error('Please pass the mandatory workflow_run_id as parameter.')
382                raise WorkflowError('Please pass the mandatory workflow_run_id as parameter.')
383            if workflow_id is None:
384                self.logger.error('Please pass the mandatory workflow id as parameter.')
385                raise WorkflowError('Please pass the mandatory workflow_id as parameter.')
386            response = IWUtils.ejson_deserialize(self.call_api("GET", url_builder.get_workflow_status_url(
387                self.client_config, workflow_id, workflow_run_id), IWUtils.get_default_header_for_v3(
388                self.client_config['bearer_token'])).content)
389            result = response.get('result', {})
390            workflow_status = result['workflow_status']["state"]
391            while workflow_status.lower() not in ['success', 'completed', 'failed', 'aborted', 'canceled']:
392                response = IWUtils.ejson_deserialize(self.call_api("GET", url_builder.get_workflow_status_url(

395                result = response.get('result', {})
396                workflow_status = result['workflow_status']["state"]
397                time.sleep(poll_interval)
398
399            run_id = result.get('id', None)

401                self.logger.error(f'Failed to poll status of the workflow {workflow_id}')
